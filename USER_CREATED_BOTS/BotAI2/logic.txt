        // ----- Three.js Immersive Background -----
        (function initThree() {
            const canvas = document.getElementById('bg-canvas');
            const scene = new THREE.Scene();
            scene.background = new THREE.Color(0x050510);
            scene.fog = new THREE.FogExp2(0x050510, 0.002);

            const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.set(0, 2, 15);

            const renderer = new THREE.WebGLRenderer({ canvas, alpha: false, antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(window.devicePixelRatio);

            // --- Lights ---
            const ambientLight = new THREE.AmbientLight(0x404060);
            scene.add(ambientLight);

            const dirLight = new THREE.DirectionalLight(0xff6666, 0.8);
            dirLight.position.set(1, 2, 4);
            scene.add(dirLight);

            const pointLight = new THREE.PointLight(0xff3300, 1, 30);
            pointLight.position.set(-3, 1, 5);
            scene.add(pointLight);

            // --- Particles (stars / embers) ---
            const particleCount = 1500;
            const particlesGeo = new THREE.BufferGeometry();
            const particlesPos = new Float32Array(particleCount * 3);
            const colors = new Float32Array(particleCount * 3);

            for (let i = 0; i < particleCount; i++) {
                // Spread in a sphere
                const r = 20 + Math.random() * 30;
                const theta = Math.random() * Math.PI * 2;
                const phi = Math.acos((Math.random() * 2) - 1);
                
                const x = r * Math.sin(phi) * Math.cos(theta);
                const y = r * Math.sin(phi) * Math.sin(theta);
                const z = r * Math.cos(phi);
                
                particlesPos[i*3] = x;
                particlesPos[i*3+1] = y * 0.5; // flatten vertically a bit
                particlesPos[i*3+2] = z;

                // Color: red/orange/white
                const colorVal = 0.8 + Math.random() * 0.5;
                colors[i*3] = 1.0;        // R
                colors[i*3+1] = 0.3 + Math.random() * 0.4; // G
                colors[i*3+2] = 0.1 + Math.random() * 0.2; // B
            }

            particlesGeo.setAttribute('position', new THREE.BufferAttribute(particlesPos, 3));
            particlesGeo.setAttribute('color', new THREE.BufferAttribute(colors, 3));

            const particleMaterial = new THREE.PointsMaterial({ 
                size: 0.15, 
                vertexColors: true,
                transparent: true,
                opacity: 0.9,
                blending: THREE.AdditiveBlending
            });

            const particles = new THREE.Points(particlesGeo, particleMaterial);
            scene.add(particles);

            // --- Floating orbs (spooky) ---
            const orbGroup = new THREE.Group();
            const orbCount = 8;
            for (let i = 0; i < orbCount; i++) {
                const geometry = new THREE.SphereGeometry(0.5 + Math.random()*0.4, 16, 16);
                const material = new THREE.MeshStandardMaterial({
                    color: new THREE.Color(`hsl(${Math.random()*20 + 340}, 100%, 50%)`),
                    emissive: new THREE.Color(0x330000),
                    roughness: 0.3,
                    metalness: 0.1
                });
                const sphere = new THREE.Mesh(geometry, material);
                
                const angle = (i / orbCount) * Math.PI * 2;
                const radius = 4 + Math.random() * 2;
                sphere.position.set(
                    Math.cos(angle) * radius,
                    Math.sin(angle*2) * 1.5,
                    Math.sin(angle) * radius * 0.5
                );
                orbGroup.add(sphere);
            }
            scene.add(orbGroup);

            // --- Additional small floating particles near camera ---
            const nearParticlesGeo = new THREE.BufferGeometry();
            const nearPos = new Float32Array(400 * 3);
            for (let i = 0; i < 400; i++) {
                nearPos[i*3] = (Math.random() - 0.5) * 20;
                nearPos[i*3+1] = (Math.random() - 0.5) * 10;
                nearPos[i*3+2] = (Math.random() - 5) * 5;
            }
            nearParticlesGeo.setAttribute('position', new THREE.BufferAttribute(nearPos, 3));
            const nearParticlesMat = new THREE.PointsMaterial({ color: 0xff6666, size: 0.08, transparent: true, opacity: 0.5, blending: THREE.AdditiveBlending });
            const nearParticles = new THREE.Points(nearParticlesGeo, nearParticlesMat);
            scene.add(nearParticles);

            // --- Animation loop ---
            let clock = new THREE.Clock();

            function animate() {
                requestAnimationFrame(animate);

                const delta = clock.getDelta();
                const elapsedTime = performance.now() * 0.001; // seconds

                // Rotate particles slowly
                particles.rotation.y += 0.0001;
                particles.rotation.x += 0.00005;

                // Move orbs in a creepy floating way
                orbGroup.children.forEach((orb, index) => {
                    orb.position.x += Math.sin(elapsedTime + index) * 0.002;
                    orb.position.y += Math.cos(elapsedTime * 0.8 + index) * 0.002;
                });

                // Rotate near particles
                nearParticles.rotation.y += 0.0005;

                // Slight camera movement (subtle)
                camera.position.x = Math.sin(elapsedTime * 0.1) * 2;
                camera.position.y = 2 + Math.sin(elapsedTime * 0.2) * 0.5;
                camera.lookAt(0, 0, 0);

                renderer.render(scene, camera);
            }

            animate();

            // Handle resize
            window.addEventListener('resize', () => {
                camera.aspect = window.innerWidth / window.innerHeight;
                camera.updateProjectionMatrix();
                renderer.setSize(window.innerWidth, window.innerHeight);
            });
        })();

        // ----- GSAP Animations for UI elements -----
        window.addEventListener('load', () => {
            // Animate header and input area on load
            gsap.from('.bot-header', { duration: 1.2, y: -100, opacity: 0, ease: 'power3.out' });
            gsap.from('.input-area', { duration: 1.2, y: 100, opacity: 0, ease: 'power3.out', delay: 0.2 });
            gsap.from('.avatar', { duration: 1, scale: 0, rotation: 360, ease: 'elastic.out(1, 0.5)', delay: 0.5 });
            
            // Subtle floating animation for the avatar
            gsap.to('.avatar', {
                duration: 3,
                y: -8,
                repeat: -1,
                yoyo: true,
                ease: 'sine.inOut'
            });

            // Pulsing glow on send button
            gsap.to('#sendBtn', {
                duration: 1.5,
                boxShadow: '0 0 35px #ff5e5e, 0 0 60px #ff0000',
                repeat: -1,
                yoyo: true,
                ease: 'sine.inOut'
            });

            // Animate each message as it appears (already handled by CSS, but we can add extra)
            // The chat messages are added dynamically; we could observe additions, but CSS fadeInUp is enough.
        });
